Input Output Equation
\[
    \phi(I_{syn})=\frac{c_{E,I}I_{syn}-I_{E,I}}{1-exp[-g_{E,I}(c_{E,I}I_{syn}-I_{E,I})]} 
\]

\section{11 Equation Model}
Selective 1
\begin{equation} \tau_{r} \frac{dr_1}{dt}=-r_1+\phi(I_{syn,1}) \end{equation}
Selective 2
\begin{equation} \tau_{r} \frac{dr_2}{dt}=-r_i+\phi(I_{syn,2}) \end{equation}
Non-selective
\begin{equation} \tau_{r} \frac{dr_{NS}}{dt}=-r_i+\phi(I_{syn,NS}) \end{equation}
Inhibitory population
\begin{equation} \tau_{r} \frac{dr_I}{dt}=-r_I+\phi(I_{syn,I}) \end{equation}
AMPA population
\begin{equation}  \frac{dS_{AMPA,1}}{dt}=-\frac{S_{AMPA,1}}{\tau_{AMPA}}+r_1 \end{equation}

\begin{equation}  \frac{dS_{AMPA,2}}{dt}=-\frac{S_{AMPA,2}}{\tau_{AMPA}}+r_2 \end{equation}
\begin{equation} \frac{dS_{AMPA,NS}}{dt}=-\frac{S_{AMPA,NS}}{\tau_{AMPA}}+r_{NS} \end{equation}

NMDA population
\begin{equation}  \frac{dS_{NMDA,1}}{dt}=-\frac{S_{NMDA,1}}{\tau_{NMDA}}+(1-S_{NMDA,1})F(\psi(r_1)) \end{equation}
Simplified as 
\[ \frac{dS_{NMDA,1}}{dt}=-\frac{S_{NMDA,1}}{\tau_{NMDA}}+(1-S_{NMDA,1})\gamma r_1 \]

\begin{equation} \tau_{r} \frac{dS_{NMDA,2}}{dt}=-\frac{S_{NMDA,2}}{\tau_{NMDA}}+(1-S_{NMDA,2})F(\psi(r_2)) \end{equation}
\begin{equation} \tau_{r} \frac{dS_{NMDA,NS}}{dt}=-\frac{S_{NMDA,NS}}{\tau_{NMDA}}+(1-S_{NMDA,NS})F(\psi(r_{NS})) \end{equation}

GABA Population
\begin{equation} \tau_{r} \frac{dS_{GABA}}{dt}=-\frac{S_{GABA}}{\tau_{GABA}}+r_I \end{equation}
